defined_text = {
	name = GetMROMajorThreatslot1Name
	text = {
			trigger = {
					tag = MRO
					check_variable = { MRO.majors_factories^0 < 0.1 }
			}
			localization_key = MRO_1_slot_low
	}
	text = {
			trigger = {
					tag = MRO
					OR = {
							AND = {
									check_variable = { MRO.majors_factories^0 > 0.1 }
									check_variable = { MRO.majors_factories^0 < 3 }
							}
							check_variable = { MRO.majors_factories^0 = 0.1 }
							check_variable = { MRO.majors_factories^0 = 3 }
					}
			}
			localization_key = MRO_1_slot_medium
	}
	text = {
			trigger = {
					tag = MRO
					check_variable = { MRO.majors_factories^0 > 3 }
			}
			localization_key = MRO_1_slot_high
	}
}
