enemy_has_equal_or_better_radar_tech_trigger = {
	custom_trigger_tooltip = {
		tooltip = enemy_has_equal_or_better_radar_tech_trigger_tt
		if = {
			limit = {
				ROOT = { 
					has_tech = basic_telepathy_II 
					NOT = {
						has_tech = telepathy_tower_I
					}
				} 
			}
			FROM = { 
				OR = {
					has_tech = basic_telepathy_II 
					has_tech = telepathy_tower_I
				}
			}
		}
		else_if = {
			limit = {
				ROOT = { 
					has_tech = telepathy_tower_I 
					NOT = {
						has_tech = telepathy_tower_II
					}
				} 
			}
			FROM = { 
				OR = {
					has_tech = telepathy_tower_I 
					has_tech = telepathy_tower_II
				}
			}
		}
		else_if = {
			limit = {
				ROOT = { 
					has_tech = telepathy_tower_II 
					NOT = {
						has_tech = telepathy_tower_III
					}
				} 
			}
			FROM = { 
				OR = {
					has_tech = telepathy_tower_II 
					has_tech = telepathy_tower_III
				}
			}
		}
		#no more steps needed because end of research tree is reached
	}
}
