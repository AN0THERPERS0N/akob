	on_annex = { 
		effect = {
			if = {
				limit = {
					country_exists = DCR
					NOT = { country_exists = ICR }
					NOT = { country_exists = LCR }
				}
				DCR = { set_country_flag = DCR_civil_war_end }
				complete_national_focus = DCR_end_of_the_civil_war
				country_event = {
					id = dcr.end_civil_war
					days = 1
					random = 1
					random_days = 0
				}
			}
			else_if = {
				limit = {
					country_exists = LCR
					NOT = { country_exists = ICR }
					NOT = { country_exists = DCR }
				}
				LCR = { set_country_flag = LCR_civil_war_end }
				complete_national_focus = LCR_peace_at_last
				country_event = {
					id = lcr.end_civil_war
					days = 1
					random = 1
					random_days = 0
				}
			}
			else_if = {
				limit = {
					country_exists = ICR
					NOT = { country_exists = DCR }
					NOT = { country_exists = LCR }
				}
				ICR = { set_country_flag = ICR_civil_war_end }
				complete_national_focus = ICR_end_of_the_civil_war
				country_event = {
					id = icr.end_civil_war
					days = 1
					random = 1
					random_days = 0
				}
			}
		}
	}
