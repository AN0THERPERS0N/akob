				}
			}

				set_politics = {
					ruling_party = fascism

					election_frequency = 48
					elections_allowed = no
				}
			}
			THB = {
				set_popularities = {
					communism = 98
					fascism = 2
				}
				
				set_politics = {
					ruling_party = fascism
					last_election = "1710.3.5"
					election_frequency = 48
					elections_allowed = no
				}
			}
		}
	}
	
	option = {
		name = dcr.end_civil_war.a
		DCR = { add_manpower = 200000 }
		ai_chance = {
			factor = 20
		}
	}

	option = {
		name = dcr.end_civil_war.b

		ai_chance = {
			factor = 10
		}
	}
