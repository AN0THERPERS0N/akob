				target = LCR

			}
			annex_country = {
				target = ICR
				transfer_troops = yes
			}
			DCR = {
				if = {
					limit = {
						has_dlc = "Together for Victory"
					}
					set_autonomy = {
						target = AGR
						autonomous_state = autonomy_integrated_puppet
					}
					set_autonomy = {
						target = THB
						autonomous_state = autonomy_integrated_puppet
					}
					else = {
						puppet = AGR
						puppet = THB
					}
				}
			}
			AGR = {
				set_popularities = {
					communism = 98
					fascism = 2
				}
				
				set_politics = {
					ruling_party = fascism
					last_election = "1710.3.5"
					election_frequency = 48
					elections_allowed = no
				}
			}
			THB = {
				set_popularities = {
					communism = 98
					fascism = 2
				}
				
				set_politics = {
					ruling_party = fascism
					last_election = "1710.3.5"
					election_frequency = 48
					elections_allowed = no
				}
			}
		}
	}
	
	option = {
		name = dcr.end_civil_war.a
		DCR = { add_manpower = 1500000 }
		ai_chance = {
			factor = 20
		}
	}

	option = {
		name = dcr.end_civil_war.b
		add_political_power = 1500
		ai_chance = {
			factor = 10
		}
	}
}
